// Function to create the dynamic bar chart function createChart(data) { const canvas = document.getElementById('chartCanvas'); if (!canvas) { console.error("Canvas element not found."); return; // Exit the function if element is not found } // Get wrapper dimensions const wrapperWidth = canvas.parentElement.offsetWidth; const wrapperHeight = canvas.parentElement.offsetHeight; // Set canvas dimensions canvas.width = wrapperWidth; canvas.height = wrapperHeight; const ctx = canvas.getContext('2d'); // Clear canvas ctx.clearRect(0, 0, canvas.width, canvas.height); const barWidth = wrapperWidth / data.length; const maxValue = Math.max(...data); const scale = wrapperHeight / maxValue; data.forEach((value, index) => { const x = index * barWidth; const height = value * scale; ctx.fillStyle = 'blue'; ctx.fillRect(x, wrapperHeight - height, barWidth, height); }); }
